def parse_args():
    parser = argparse.ArgumentParser(description='Copy specific files from SLURM to local directory')
    parser.add_argument('--exp-id', type=str, required=True, 
                        help='Experiment ID (e.g., 20250401_181800)')
    parser.add_argument('--user', type=str, required=True, 
                        help='Username for SLURM server')
    parser.add_argument('--server', type=str, default='cs-oci-ord-dc-02', 
                        help='SLURM server hostname')
    parser.add_argument('--account', type=str, default='nvr_torontoai_humanmotionfm',
                        help='Account/project name on SLURM cluster')
    # NOTE: need to use fs12 (the actual storage) instead of fsw
    parser.add_argument('--remote-base', type=str, default=None,
                        help='Base path on remote server (overrides automatic path construction)')
    parser.add_argument('--local-base', type=str, default='./exps',
                        help='Local base directory to save files')
    
    # Download options
    parser.add_argument('--download-last', action='store_true', 
                        help='Download last.ckpt checkpoint')
    parser.add_argument('--download-score-based', action='store_true', 
                        help='Download score_based.ckpt checkpoint')
    parser.add_argument('--download-epoch', type=int, default=None,
                        help='Download closest epoch_xxx.ckpt to specified epoch number')
    parser.add_argument('--download-motion-lib-epoch', type=int, default=None,
                        help='Download closest predicted_motion_lib_epoch_xxx.pt to specified epoch')
    parser.add_argument('--download-env', action='store_true',
                        help='Download environment checkpoint files (env_*.ckpt)')
    parser.add_argument('--skip-config', action='store_true', default=False,
                        help='Skip downloading config files (config.yaml and resolved_configs*.pt)')
    
    return parser.parse_args()

def run_ssh_command(ssh_prefix, command):
    """Run SSH command and return output"""
    cmd = f"{ssh_prefix} '{command}'"
    result = subprocess.run(cmd, shell=True, capture_output=True, text=True)
    return result.stdout.strip()

def list_remote_files(ssh_prefix, remote_path, pattern="*"):
    """List files matching pattern in remote path"""
    cmd = f"find {remote_path} -maxdepth 1 -name '{pattern}' -type f"
    output = run_ssh_command(ssh_prefix, cmd)
    return [line.strip() for line in output.splitlines() if line.strip()]

def find_closest_epoch_file(ssh_prefix, remote_dir, target_epoch, file_pattern):
    """
    Find the closest epoch-numbered file to target_epoch.
    
    Args:
        ssh_prefix: SSH command prefix
        remote_dir: Directory to search in
        target_epoch: Target epoch number
        file_pattern: Pattern like "epoch_*.ckpt" or "predicted_motion_lib_epoch_*.pt"
    
    Returns:
        Tuple of (closest_file_path, epoch_number) or (None, None) if not found
    """
    files = list_remote_files(ssh_prefix, remote_dir, file_pattern)
    
    if not files:
        return None, None
    
    # Extract epoch numbers from filenames
    epoch_pattern = r'epoch_(\d+)'
    file_epochs = []
    for f in files:
        match = re.search(epoch_pattern, f)
        if match:
            epoch_num = int(match.group(1))
            file_epochs.append((f, epoch_num))
    
    if not file_epochs:
        return None, None
    
    # Find closest epoch
    closest = min(file_epochs, key=lambda x: abs(x[1] - target_epoch))
    return closest[0], closest[1]

def find_results_subdirectories(ssh_prefix, remote_exp_path):
    """
    Find all result subdirectories matching pattern results/*/
    
    Returns:
        List of directory paths
    """
    cmd = f"find {remote_exp_path}/results -mindepth 1 -maxdepth 1 -type d"
    output = run_ssh_command(ssh_prefix, cmd)
    return [line.strip() for line in output.splitlines() if line.strip()]

def download_file(source, dest, user, server):
    """Download a single file from remote to local"""
    # Ensure destination directory exists
    os.makedirs(os.path.dirname(dest), exist_ok=True)
    
    # Build rsync command
    cmd = ['rsync', '-avz', '--partial', '--chmod=775', f"{user}@{server}:{source}", dest]
    
    print(f"Downloading: {source} -> {dest}")
    subprocess.run(cmd, check=True)

def main():
    args = parse_args()
    
    # Construct paths
    exp_id = args.exp_id.replace('exp-', '')  # Strip 'exp-' prefix if present
    exp_dir = f"exp-{exp_id}"
    
    # Build remote base path if not explicitly provided
    if args.remote_base is None:
        remote_base = f"/lustre/fs12/portfolios/nvr/projects/{args.account}/workspaces/physics-based/exps/"
    else:
        remote_base = args.remote_base
    
    remote_exp_path = f"{remote_base}{args.user}/{exp_dir}"
    local_base_dir = Path(args.local_base)
    local_exp_dir = local_base_dir / exp_dir
    
    # Create local directory if it doesn't exist
    os.makedirs(local_exp_dir, exist_ok=True)
    
    # SSH prefix for remote commands
    ssh_prefix = f"ssh {args.user}@{args.server}"
    
    print(f"Downloading files from experiment {exp_id}...")
    print(f"Remote: {args.user}@{args.server}:{remote_exp_path}")
    print(f"Local: {local_exp_dir}")
    print()
    
    files_to_download = []
    
    # Find results subdirectories (where ALL files are stored on remote)
    print("Searching for results subdirectories...")
    results_dirs = find_results_subdirectories(ssh_prefix, remote_exp_path)
    
    if not results_dirs:
        print("âœ— No results subdirectories found - no files available to download")
    else:
        print(f"Found {len(results_dirs)} results subdirectory(ies)")
        
        # Use the first results subdirectory for all downloads
        # (typically there's only one active results dir per experiment)
        results_dir = results_dirs[0]
        rel_path = results_dir.replace(f"{remote_exp_path}/", "")
        print(f"Using: {rel_path}")
        
        if len(results_dirs) > 1:
            print(f"Note: Multiple results directories found. Using first one: {rel_path}")
            print(f"      Other directories: {[d.replace(f'{remote_exp_path}/', '') for d in results_dirs[1:]]}")
        print()
        
        # Download config files from results subdirectory to local exp root
        if not args.skip_config:
            # Download config.yaml
            config_file = f"{results_dir}/config.yaml"
            exists = run_ssh_command(ssh_prefix, f"test -f {config_file} && echo 'exists'")
            if exists:
                files_to_download.append((config_file, local_exp_dir / "config.yaml"))
                print("âœ“ Will download: config.yaml")
            else:
                print("âœ— config.yaml not found")
            
            # Download all resolved_configs*.pt files
            resolved_configs_files = list_remote_files(ssh_prefix, results_dir, "resolved_configs*.*")
            for resolved_config_file in resolved_configs_files:
                filename = os.path.basename(resolved_config_file)
                files_to_download.append((resolved_config_file, local_exp_dir / filename))
                print(f"âœ“ Will download: {filename}")
            if not resolved_configs_files:
                print("âœ— No resolved_configs*.pt files found")
        
        # Download last.ckpt from results subdirectory to local exp root
        if args.download_last:
            last_file = f"{results_dir}/last.ckpt"
            exists = run_ssh_command(ssh_prefix, f"test -f {last_file} && echo 'exists'")
            if exists:
                files_to_download.append((last_file, local_exp_dir / "last.ckpt"))
                print("âœ“ Will download: last.ckpt")
            else:
                print("âœ— last.ckpt not found")
        
        # Download score_based.ckpt from results subdirectory to local exp root
        if args.download_score_based:
            score_file = f"{results_dir}/score_based.ckpt"
            exists = run_ssh_command(ssh_prefix, f"test -f {score_file} && echo 'exists'")
            if exists:
                files_to_download.append((score_file, local_exp_dir / "score_based.ckpt"))
                print("âœ“ Will download: score_based.ckpt")
            else:
                print("âœ— score_based.ckpt not found")
        
        # Download closest epoch checkpoint from results subdirectory to local exp root
        if args.download_epoch is not None:
            epoch_file, actual_epoch = find_closest_epoch_file(
                ssh_prefix, results_dir, args.download_epoch, "epoch_*.ckpt"
            )
            if epoch_file:
                filename = os.path.basename(epoch_file)
                files_to_download.append((epoch_file, local_exp_dir / filename))
                print(f"âœ“ Will download: {filename} (requested epoch {args.download_epoch}, found {actual_epoch})")
            else:
                print(f"âœ— No epoch checkpoints found near epoch {args.download_epoch}")
        
        # Download environment checkpoints
        if args.download_env:
            env_files = list_remote_files(ssh_prefix, results_dir, "env_*.ckpt")
            for env_file in env_files:
                filename = os.path.basename(env_file)
                files_to_download.append((env_file, local_exp_dir / filename))
                print(f"âœ“ Will download: {filename}")
            if not env_files:
                print("âœ— No environment checkpoints found")
        
        # Download closest predicted motion lib to local exp root
        if args.download_motion_lib_epoch is not None:
            motion_lib_file, actual_epoch = find_closest_epoch_file(
                ssh_prefix, results_dir, args.download_motion_lib_epoch, "predicted_motion_lib_epoch_*.pt"
            )
            if motion_lib_file:
                filename = os.path.basename(motion_lib_file)
                files_to_download.append((motion_lib_file, local_exp_dir / filename))
                print(f"âœ“ Will download: {filename} (requested epoch {args.download_motion_lib_epoch}, found {actual_epoch})")
            else:
                print(f"âœ— No predicted motion lib found near epoch {args.download_motion_lib_epoch}")
    
    # Download all identified files
    if files_to_download:
        print(f"\nDownloading {len(files_to_download)} file(s)...")
        print()
        for remote_path, local_path in files_to_download:
            download_file(remote_path, str(local_path), args.user, args.server)
        print()
        print(f"âœ“ Successfully downloaded {len(files_to_download)} file(s) to {local_exp_dir}")
    else:
        print("\nâœ— No files to download. Please specify what to download with flags.")
        print("   Use --help to see available options.")




if __name__ == "__main__":
    main()
